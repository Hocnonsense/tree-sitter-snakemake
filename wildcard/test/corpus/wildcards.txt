===============================================================================
Plain text
===============================================================================

a

-------------------------------------------------------------------------------

(source
  (text))

===============================================================================
Simple wildcard
===============================================================================

{a}

-------------------------------------------------------------------------------

(source
  (text)
  (wildcard
    name: (name))
  (text))

===============================================================================
Wildcard with constraint
===============================================================================

{a,\d+}

-------------------------------------------------------------------------------

(source
  (text)
  (wildcard
    name: (name)
    constraint: (constraint))
  (text))

===============================================================================
Wildcard: constraint containing {N} quantifier
===============================================================================

{a,x\d{3}}{b}

-------------------------------------------------------------------------------

(source
  (text)
  (wildcard
    name: (name)
    constraint: (constraint))
  (wildcard
    name: (name))
  (text))

===============================================================================
Wildcard with format flag
===============================================================================

{input:q}

-------------------------------------------------------------------------------

(source
  (text)
  (wildcard
    name: (name)
    flag: (flag))
  (text))
